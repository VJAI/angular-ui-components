Dialog

Overview

Lego provides built-in dialogs like alert and confirm. You can also create custom dialogs.

Alert Dialog

Open Alert Dialog 
<a href="#" (click)="openAlert($event)">Open Alert Dialog</a>

constructor(private dialog: DialogService) {
}

public openAlert(event: UIEvent) {
  event.preventDefault();
  this.dialog.alert('Success', 'The changes to the post saved successfully!', MessageType.Success);
}

Confirm Dialog

Open Confirm Dialog 
<a href="#" (click)="openConfirm($event)">Open Confirm Dialog</a>

constructor(private dialog: DialogService) {
}

public openConfirm(event: UIEvent) {
  event.preventDefault();
  this.dialog.confirm('Confirm Approval', 'Are you sure want to approve this contract?');
}

Open Custom Dialog

}

public openCustom(event: UIEvent) {
